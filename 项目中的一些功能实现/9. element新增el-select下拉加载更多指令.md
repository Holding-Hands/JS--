#### element新增el-select下拉加载更多指令

* 首先需要在mainjs引入下面代码

```js
import Vue from 'vue'
// 滚动加载更多
Vue.directive('loadMore', {
    bind(el, binding) {
        // 获取element，定义scroll
        let select_dom = el.querySelector('.el-select-dropdown .el-select-dropdown__wrap')
        select_dom.addEventListener('scroll', function () {
            let height = this.scrollHeight - this.scrollTop <= this.clientHeight
            if (height) {
                binding.value()
            }
        })
    }
})
```



* 在el-select中使用

```html
						<el-select
                                clearable
                                filterable
                                v-model="name"
                                :remote-method="remoteMethod"
                                @visible-change="(...arg) => onVisibleChange(...arg, item)"
                                remote
                                :loading="loading"
                                v-loadMore="() => onLoadMore()"
                            >
                                <el-option
                                    v-for="itemY in emissionLawList"
                                    :key="itemY.id"
                                    :label="itemY.name"
                                    :value="itemY.name"

                            </el-select>
```
